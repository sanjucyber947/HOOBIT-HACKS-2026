package com.example.myapplication // Make sure this matches your package name!

import android.content.Context
import android.graphics.Typeface
import android.os.Bundle
import android.view.inputmethod.EditorInfo
import android.view.inputmethod.InputMethodManager
import android.widget.Button
import android.widget.EditText
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // 1. Create a vertical layout dynamically so we don't mess up your XML
        val layout = android.widget.LinearLayout(this).apply {
            orientation = android.widget.LinearLayout.VERTICAL
            gravity = android.view.Gravity.CENTER
            setPadding(50, 50, 50, 50)
        }

        // 2. Create the Rounded Input Box with BIG FONT
        val inputField = EditText(this).apply {
            hint = "Type your prompt here..."
            setSingleLine(true)
            imeOptions = EditorInfo.IME_ACTION_SEND
            
            // --- BIG FONT CUSTOMIZATION ---
            textSize = 28f // Increased font size significantly
            setTypeface(null, Typeface.BOLD) // Made the text bold
            // ------------------------------

            // Apply quick programmatical rounding & padding
            val drawable = android.graphics.drawable.GradientDrawable().apply {
                shape = android.graphics.drawable.GradientDrawable.RECTANGLE
                cornerRadius = 24f // Rounded corners
                setColor(android.graphics.Color.parseColor("#F1F1F1")) // Light grey background
                setStroke(2, android.graphics.Color.parseColor("#CCCCCC")) // Grey border
            }
            background = drawable
            setPadding(40, 40, 40, 40) // Increased internal padding for the larger text
            
            // Layout margins
            layoutParams = android.widget.LinearLayout.LayoutParams(
                android.widget.LinearLayout.LayoutParams.MATCH_PARENT,
                android.widget.LinearLayout.LayoutParams.WRAP_CONTENT
            ).apply {
                bottomMargin = 40
            }
        }

        // 3. Create your Button
        val sendButton = Button(this).apply {
            text = "Send Request"
            textSize = 20f // Made the button font a bit larger too
            layoutParams = android.widget.LinearLayout.LayoutParams(
                android.widget.LinearLayout.LayoutParams.MATCH_PARENT,
                android.widget.LinearLayout.LayoutParams.WRAP_CONTENT
            )
        }

        // 4. Function to trigger the backend request
        fun sendData() {
            val text = inputField.text.toString().trim()
            if (text.isEmpty()) {
                Toast.makeText(this@MainActivity, "Please type something first!", Toast.LENGTH_SHORT).show()
                return
            }
            
            // Hide the software keyboard safely
            val imm = getSystemService(Context.INPUT_METHOD_SERVICE) as InputMethodManager
            imm.hideSoftInputFromWindow(inputField.windowToken, 0)

            Toast.makeText(this@MainActivity, "Sending: $text", Toast.LENGTH_SHORT).show()
            
            // Execute the network request using your Retrofit network module
            // RetrofitClient.instance.sendPrompt(text)
        }

        // 5. Handle Keyboard "Enter" key action
        inputField.setOnEditorActionListener { _, actionId, _ ->
            if (actionId == EditorInfo.IME_ACTION_SEND || actionId == EditorInfo.IME_ACTION_UNSPECIFIED) {
                sendData()
                true
            } else {
                false
            }
        }

        // 6. Handle Button Click
        sendButton.setOnClickListener {
            sendData()
        }

        // Add views to layout and set the screen content
        layout.addView(inputField)
        layout.addView(sendButton)
        setContentView(layout)
    }
}
